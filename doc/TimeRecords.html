def grouped_by_started_at
  uniques = map(&:started_at).uniq
  uniques = map { |time_record| time_record.started_at.to_date }.uniq

  uniques.map do |started_at|
    same = select { |record| record.started_at.to_date == started_at.to_date }

    self.class.new(same)
  end
end
